Standardization of Document Style〜文書スタイルの適用パターン〜

2014-07-22に英国政府における公文書として，閲覧用にPDF/AかHTML，共同作業用にODFを採用するとの指針が発表された。これにより，ODFの利用が促進されることが考えられる。しかし，現実としては組織内文書としてはMS Officeで採用されているOOXMLの数は無視できない。また，Web上に公開するに置いてはHTMLを使わざるを得ない。その他，MarkdownやreStructuredText，AsciiDoc，Wiki記法といった軽量マークアップにより文書が作成される事例も存在している。


近年においては，CSS組版と呼ばれるHTML+CSSによりWebページだけでなく書籍を作成するという試みが見られている。今後Web文書と紙文書が単一の文書から生成される展望が見えつつある。

申し込み時の発表概要

2014-07-22に英国政府における公文書としてODFを採用する方針が発表された。これによりLibreOfficeの利用促進が期待される。しかし，現実にはMS Officeの標準形式であるOOXMLの利用も多い。また文書をWebページに公開する上で，文書形式としてのHTMLも無視できない。そこで，LibreOffice WriterとMS Wordに存在する標準スタイルと，HTMLでの対応を確認し，相互運用性を高めるスタイルの利用方法を検討する。

Sub oWriterStyle
  	Dim oDoc
  	Dim Dummy()
  		On Error Goto oBad
		oDoc = StarDesktop.loadComponentFromURL("private:factory/swriter", "_blank", 0, Dummy())
		'Get the StyleFamilies
			Dim oFamilies
			Dim oFamilyNames
			Dim oStyleName
				oFamilies = oDoc.StyleFamilies
				oFamilyNames = oFamilies.getElementNames()
				oStyleName = oFamilies.getByName("CharacterStyles") 
				oSElementName = oStyleName.ElementNames
				oDisp = ""
			'Get the Style Name
				for i = LBound(oSElementName) to UBound(oSElementName)
					oDisp = oDisp & i & ")" & oSElementName(i)
					oDisp = oDisp & Chr$(10)
				next i
			msgbox(oDisp, 0, "Style Name")
		Exit Sub
	oBad:
		Dim oErLine As Integer
		Dim oErNum As Integer
		Dim oErMsg As String
			oErLine = Erl
			oErNum = Err
			oErMsg = Error
		Msgbox("Error Line No. " & Chr$(9) & " : " & oErLine & Chr$(10) _
			& "Error Number " & Chr$(9) & " : " & oErNum &Chr$(10 ) _
			& "Error Message" & Chr$(9) & " : " & oErMsg , 0, "Error Message")
End Sub
